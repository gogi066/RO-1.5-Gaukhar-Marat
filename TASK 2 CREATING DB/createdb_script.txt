DROP DATABASE IF EXISTS online_learning_system;
CREATE DATABASE online_learning_system;
USE online_learning_system;

-- 1 Users table
CREATE TABLE users(
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(50) NOT NULL,
    name_length INT GENERATED ALWAYS AS (CHAR_LENGTH(full_name)) STORED, -- GENERATED ALWAYS AS: automatically calculates length of full name
    birth_date DATE NOT NULL,
    gender ENUM('male','female') NOT NULL,
    email VARCHAR(50) NOT NULL UNIQUE,
    phone VARCHAR(15) UNIQUE,
    registration_date DATE NOT NULL CHECK (registration_date > '2026-01-01') -- CHECK: registration_date must be after 2026-01-01
);

-- 2 Courses table
CREATE TABLE courses(
    course_id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(50) NOT NULL,
    description TEXT,
    duration_weeks INT NOT NULL CHECK (duration_weeks > 0), -- CHECK: duration must be positive
    created_at DATE NOT NULL
);

-- 3 Instructors table
CREATE TABLE instructors(
    instructor_id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(50) NOT NULL,
    hire_date DATE NOT NULL,
    email VARCHAR(50) NOT NULL UNIQUE,
    salary DECIMAL(8,2) NOT NULL CHECK (salary > 0) -- CHECK: salary must be positive
);

-- 4 Enrollments table
CREATE TABLE enrollments(
    enroll_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    course_id INT NOT NULL,
    instructor_id INT NOT NULL,
    enroll_date DATE NOT NULL,
    progress_percent DECIMAL(5,2) DEFAULT 0 CHECK (progress_percent BETWEEN 0 AND 100), -- CHECK: progress must be between 0 and 100
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(course_id) ON DELETE CASCADE,
    FOREIGN KEY (instructor_id) REFERENCES instructors(instructor_id) ON DELETE CASCADE
);

-- 5 Lessons table
CREATE TABLE lessons(
    lesson_id INT AUTO_INCREMENT PRIMARY KEY,
    course_id INT NOT NULL,
    instructor_id INT NOT NULL,
    lesson_title VARCHAR(50) NOT NULL,
    lesson_date DATE CHECK (lesson_date > '2026-01-01'), -- CHECK: date after 2026
    duration_minutes INT CHECK (duration_minutes > 0), -- CHECK: duration positive
    FOREIGN KEY (course_id) REFERENCES courses(course_id) ON DELETE CASCADE,
    FOREIGN KEY (instructor_id) REFERENCES instructors(instructor_id) ON DELETE CASCADE
);

-- 6 Assessments table
CREATE TABLE assessments(
    assessment_id INT AUTO_INCREMENT PRIMARY KEY,
    lesson_id INT NOT NULL,
    title VARCHAR(50) NOT NULL,
    max_score INT NOT NULL CHECK (max_score > 0), -- CHECK: max_score must be positive
    FOREIGN KEY (lesson_id) REFERENCES lessons(lesson_id) ON DELETE CASCADE
);

-- 7 Lesson Progress table
CREATE TABLE lesson_progress(
    lesson_progress_id INT AUTO_INCREMENT PRIMARY KEY,
    lesson_id INT NOT NULL,
    assessment_id INT NOT NULL,
    score INT CHECK (score >= 0), -- CHECK: score cannot be negative
    FOREIGN KEY (lesson_id) REFERENCES lessons(lesson_id) ON DELETE CASCADE,
    FOREIGN KEY (assessment_id) REFERENCES assessments(assessment_id) ON DELETE CASCADE
);

-- 8 Feedback table
CREATE TABLE feedback(
    feedback_id INT AUTO_INCREMENT PRIMARY KEY,
    lesson_progress_id INT NOT NULL,
    comment TEXT,
    rating INT CHECK (rating BETWEEN 1 AND 5),
    feedback_date DATE NOT NULL,
    FOREIGN KEY (lesson_progress_id) REFERENCES lesson_progress(lesson_progress_id) ON DELETE CASCADE
);

-- 9 Discussions table
CREATE TABLE discussions(
    discussion_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    lesson_id INT NOT NULL,
    message TEXT NOT NULL,
    posted_at DATETIME NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (lesson_id) REFERENCES lessons(lesson_id) ON DELETE CASCADE
);

-- 10 Certificates table
CREATE TABLE certificates(
    certificate_id INT AUTO_INCREMENT PRIMARY KEY,
    enroll_id INT NOT NULL,
    issue_date DATE NOT NULL CHECK (issue_date > '2026-01-01'),
    FOREIGN KEY (enroll_id) REFERENCES enrollments(enroll_id) ON DELETE CASCADE
);
