CREATE SCHEMA IF NOT EXISTS online_learning_db;
SET search_path TO online_learning_db;

CREATE TABLE IF NOT EXISTS Users (
    user_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    full_name VARCHAR(101) GENERATED ALWAYS AS (first_name  ' '  last_name) STORED,
    email VARCHAR(100) UNIQUE NOT NULL,
    gender CHAR(1) CHECK (gender IN ('M', 'F', 'O')),
    registration_date DATE DEFAULT CURRENT_DATE CHECK (registration_date > '2026-01-01')
);

CREATE TABLE IF NOT EXISTS Instructors (
    instructor_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    full_name VARCHAR(100) NOT NULL,
    expertise VARCHAR(100),
    bio TEXT
);

CREATE TABLE IF NOT EXISTS Courses (
    course_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    instructor_id INT REFERENCES Instructors(instructor_id) ON DELETE SET NULL,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    price DECIMAL(10, 2) DEFAULT 0 CHECK (price >= 0)
);

CREATE TABLE IF NOT EXISTS Lessons (
    lesson_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    course_id INT REFERENCES Courses(course_id) ON DELETE CASCADE,
    title VARCHAR(255) NOT NULL,
    content_url TEXT,
    duration_minutes INT CHECK (duration_minutes > 0)
);

CREATE TABLE IF NOT EXISTS Enrollments (
    enrollment_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id INT REFERENCES Users(user_id) ON DELETE CASCADE,
    course_id INT REFERENCES Courses(course_id) ON DELETE CASCADE,
    enrollment_date DATE DEFAULT CURRENT_DATE CHECK (enrollment_date > '2026-01-01'),
    progress_percent DECIMAL(5, 2) DEFAULT 0 CHECK (progress_percent BETWEEN 0 AND 100),
    status VARCHAR(20) DEFAULT 'Active' CHECK (status IN ('Active', 'Completed', 'Dropped'))
);

CREATE TABLE IF NOT EXISTS Assessments (
    assessment_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    course_id INT REFERENCES Courses(course_id) ON DELETE CASCADE,
    title VARCHAR(255) NOT NULL,
    max_score INT DEFAULT 100 CHECK (max_score > 0)
);

CREATE TABLE IF NOT EXISTS Learning_Progress (
    progress_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id INT REFERENCES Users(user_id) ON DELETE CASCADE,
    lesson_id INT REFERENCES Lessons(lesson_id) ON DELETE CASCADE,
    is_completed BOOLEAN DEFAULT FALSE,
    completion_date TIMESTAMP
);

CREATE TABLE IF NOT EXISTS Certifications (
    certificate_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id INT REFERENCES Users(user_id) ON DELETE CASCADE,
    course_id INT REFERENCES Courses(course_id) ON DELETE CASCADE,
    issue_date DATE CHECK (issue_date > '2026-01-01')
);

CREATE TABLE IF NOT EXISTS Feedback (
    feedback_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id INT REFERENCES Users(user_id) ON DELETE CASCADE,
    course_id INT REFERENCES Courses(course_id) ON DELETE CASCADE,
    rating INT CHECK (rating BETWEEN 1 AND 5),
    comment TEXT,
    feedback_date DATE DEFAULT CURRENT_DATE
);

TRUNCATE TABLE 
    Feedback, 
    Certifications, 
    Learning_Progress, 
    Assessments, 
    Enrollments, 
    Lessons, 
    Courses, 
    Instructors, 
    Users 
RESTART IDENTITY CASCADE;
